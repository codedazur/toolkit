# usePrevious

Returns the previous value of a variable.

## Usage

```tsx
const [position, setPosition] = useState(1);
const previousPosition = usePrevious(position);

// set position to 2 on first render
useEffect(() => setPosition(2), []);

// will log undefined on first render
// will log 1 on second render
useEffect(() => {
  console.log(deltaPosition);
}, [deltaPosition]);
```

## Use-Case: Stepper

```tsx
import { useState } from "react";
import { sequence } from "@codedazur/essentials";
import { usePrevious } from "@codedazur/react-essentials";

export function Stepper({ count }) {
  const [currentStep, setCurrentStep] = useState(0);
  const previousStep = usePrevious(currentStep);

  return (
    <Row>
      {sequence(1, count).map((step) => (
        <Step isActive={currentStep === step} wasActive={previousStep === step}>
          {step}
        </Step>
      ))}
    </Row>
  );
}
```
